362  # read = Number of bytes to read from the SPI device
363  #
364  def Transaction(self,OutBuffer, read = 0):
365  ReceiveBuffer = create_string_buffer(chr(0) * 0x80)
366  TransmitBuffer= create_string_buffer(OutBuffer)
367  Transaction = spi_ioc_transfer()
368  Transaction.speed_hz = c_uint32(self.Speed)
369  Transaction.tx_buf=addressof(TransmitBuffer)
370  Transaction.rx_buf=addressof(ReceiveBuffer)
371  Transaction.delay_usecs = self.Delay
372  if read > 0 and self.Speed!= self.ReadSpeed: # Slow down speed for reading
373  Transaction.speed_hz = self.ReadSpeed
374  elif read==0 and self.Speed!=self.WriteSpeed:
375  Transaction.speed_hz = self.WriteSpeed
376  if self.Speed != Transaction.speed_hz:
377  self.Speed = Transaction.speed_hz
378  self.SetSpeed()
379  if read > len(OutBuffer):
380  Transaction.len=read
381  else:
382  Transaction.len= len(OutBuffer)
383  Transaction.bits_per_word = self.Bits
384  Transaction.cs_change = 0
385  Transaction.pad = 0
386  ret = ioctl(self.File,SPI_IOC_MESSAGE(1), addressof(Transaction))

805 #
806 # Class For BitWizard boards with Digital/Analog IO, On board Temperature sensors etc. Use this as a baseclass for creating new board classes only.
807 # It is part of the automatic reconfiguration code. Classed using this as a baseclass can override PinConfig, ADSamples and ADBitshift. and calls to SetPinConfig and InitAnalog might be usefull and can be done in realtime.
808 # @brief class for BitWizzard boards with IOPins.
809 #
810 class IOPinBase():
811  IODevice=_ATTiny
812  PinConfig = {}
813  IOPins = 0
814  ADCChannels = 0
815  ADCChannelConfig = {}
816  ADSamples = 4096
817  ADBitshift = 6
818 
819  def __init__(self):
820  self.Pins={}
821  for p in range(self.IOPins):
822  if self.PinConfig.has_key(p):
823  self.Pins[p]=self.PinConfig[p]["device"](p,self)
824  else:
825  self.Pins[p]= DigitalIn(p,self)
826  self.DoPinConfig()
827  self.InitAnalog(self.ADSamples,self.ADBitshift)
828 
829  def DoPinConfig(self):
830  mask = 0
831  pwmmask = 0
832  for p in range(self.IOPins):
833  mask = mask | self.Pins[p].PinMask()
834  if isinstance(self.Pins[p],PWMOut):
835  pwmmask = pwmmask | 2**p
836  self.SetInputOutput(mask,pwmmask)
837 
838  ##
839  #
840  # Set or change the functionality of one IOPin.
841  # @param pin Int: The Pinnummer on the board to Set/Change.
842  # @param pintype class: Can be any subclass of IOPin, like DigitalIn, DogitalOut, PWMOut, AnalogIn, MCP9700 etc.
843  # @param kwargs list: additional configuration passed to the __init__ when in instance of pintype is created.
844  #
845  def SetPinConfig(self,pin, pintype, **kwargs):
846  if self.Pins[pin]!=None:
847  del self.Pins[pin]
848  if self.PinConfig.has_key(pin):
849  if self.PinConfig[pin].has_key('property'):
850  delattr(self,self.PinConfig[pin]['property'])
851  del self.PinConfig[pin]
852  if not self.PinConfig.has_key(pin):
853  self.PinConfig[pin] = kwargs
854  self.PinConfig[pin]["device"]=pintype
855  c = self.ADCChannels
856  self.Pins[pin]=pintype(pin,self)
857  self.DoPinConfig()
858  if c!= self.ADCChannels:
859  self.Bus.Write_uInt8(self.Address,0x80,self.ADCChannels)
860
